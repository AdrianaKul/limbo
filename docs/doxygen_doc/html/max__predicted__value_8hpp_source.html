75  template <typename BO, typename AggregatorFunction>
76  bool operator()(const BO& bo, const AggregatorFunction& afun) const
77  {
78  // Prevent instantiation of GPMean if there are no observed samples
79  if (bo.observations().size() == 0)
80  return false;
81 
82  auto optimizer = _get_optimizer(typename BO::acqui_optimizer_t(), Optimizer());
83  Eigen::VectorXd starting_point = tools::random_vector(bo.model().dim_in());
84  auto model_optimization =
85  [&](const Eigen::VectorXd& x, bool g) { return opt::no_grad(afun(bo.model().mu(x))); };
86  auto x = optimizer(model_optimization, starting_point, true);
87  double val = afun(bo.model().mu(x));
88 
89  if (bo.observations().size() == 0 || afun(bo.best_observation(afun)) <= Params::stop_maxpredictedvalue::ratio() * val)
90  return false;
91  else {
92  std::cout << "stop caused by Max predicted value reached. Threshold: "
93  << Params::stop_maxpredictedvalue::ratio() * val
94  << " max observations: " << afun(bo.best_observation(afun)) << std::endl;
95  return true;
96  }
97  }
