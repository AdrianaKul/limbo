74  struct MeanLFOptimization {
75  public:
76  MeanLFOptimization(const GP& gp) : _original_gp(gp)
77  {
78  size_t n = gp.obs_mean().rows();
79 
80  // K^{-1} using Cholesky decomposition
81  _K = Eigen::MatrixXd::Identity(n, n);
82  gp.matrixL().template triangularView<Eigen::Lower>().solveInPlace(_K);
83  gp.matrixL().template triangularView<Eigen::Lower>().transpose().solveInPlace(_K);
84  }
85 
86  opt::eval_t operator()(const Eigen::VectorXd& params, bool compute_grad) const
87  {
88  GP gp(this->_original_gp);
89  gp.mean_function().set_h_params(params);
90 
91  gp.recompute(true, false);
92 
93  size_t n = gp.obs_mean().rows();
94 
95  // --- cholesky ---
96  // see:
97  // http://xcorr.net/2008/06/11/log-determinant-of-positive-definite-matrices-in-matlab/
98  Eigen::MatrixXd l = gp.matrixL();
99  long double det = 2 * l.diagonal().array().log().sum();
100 
101  double a = (gp.obs_mean().transpose() * gp.alpha())
102  .trace(); // generalization for multi dimensional observation
103  // std::cout<<" a: "<<a <<" det: "<< det<<std::endl;
104  double lik = -0.5 * a - 0.5 * det - 0.5 * n * log(2 * M_PI);
105 
106  if (!compute_grad)
107  return opt::no_grad(lik);
108 
109  Eigen::VectorXd grad = Eigen::VectorXd::Zero(params.size());
110  for (int i_obs = 0; i_obs < gp.dim_out(); ++i_obs)
111  for (size_t n_obs = 0; n_obs < n; n_obs++) {
112  grad.tail(gp.mean_function().h_params_size()) += gp.obs_mean().col(i_obs).transpose() * _K.col(n_obs) * gp.mean_function().grad(gp.samples()[n_obs], gp).row(i_obs);
113  }
114 
115  return {lik, grad};
116  }

120  Eigen::MatrixXd _K;
121  };
122  };
