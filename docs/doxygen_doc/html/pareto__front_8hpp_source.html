59  using pareto_t = std::vector<pareto_point_t>;

61  template <typename BO, typename AggregatorFunction>
62  void operator()(const BO& bo, const AggregatorFunction&)
63  {
64  if (!bo.stats_enabled() || bo.observations().empty())
65  return;
66  std::string fname = bo.res_dir() + "/" + "pareto_front_" + std::to_string(bo.current_iteration()) + ".dat";
67  std::ofstream ofs(fname.c_str());
68  auto pareto = _pareto_data(bo);
69  for (auto x : pareto) {
70  ofs << std::get<0>(x).transpose() << " "
71  << std::get<1>(x).transpose() << std::endl;
72  }
73  }

75  protected:
76  template <typename BO>
77  pareto_t _pareto_data(const BO& bo)
78  {
79  std::vector<Eigen::VectorXd> v(bo.samples().size());
80  size_t dim = bo.observations().size();
81  std::fill(v.begin(), v.end(), Eigen::VectorXd::Zero(dim));
82  return pareto::pareto_set<1>(
83  _pack_data(bo.samples(), bo.observations(), v));
84  }
85  pareto_t _pack_data(const std::vector<Eigen::VectorXd>& points,
86  const std::vector<Eigen::VectorXd>& objs,
87  const std::vector<Eigen::VectorXd>& sigma) const
88  {
89  assert(points.size() == objs.size());
90  assert(sigma.size() == objs.size());
91  pareto_t p(points.size());
92  tools::par::loop(0, p.size(), [&](size_t k) {
93  // clang-format off
94  p[k] = std::make_tuple(points[k], objs[k], sigma[k]);
95  // clang-format on
96  });
97  return p;
98  }
99  };
100  }
101  }
