139  template <typename StateFunction, typename AggregatorFunction = FirstElem>
140  void optimize(const StateFunction& sfun, const AggregatorFunction& afun = AggregatorFunction(), bool reset = true)
141  {
142 
143  this->_init(sfun, afun, reset);
144 
145  if (!this->_observations.empty())
146  _model.compute(this->_samples, this->_observations, Eigen::VectorXd::Constant(this->_observations.size(), Params::bayes_opt_boptimizer::noise()));
147  else
148  _model = model_t(StateFunction::dim_in, StateFunction::dim_out);
149 
150  acqui_optimizer_t acqui_optimizer;
151 
152  while (!this->_stop(*this, afun)) {
153  acquisition_function_t acqui(_model, this->_current_iteration);
154 
155  auto acqui_optimization =
156  [&](const Eigen::VectorXd& x, bool g) { return acqui(x,afun,g); };
157  Eigen::VectorXd starting_point = tools::random_vector(StateFunction::dim_in, Params::bayes_opt_bobase::bounded());
158  Eigen::VectorXd new_sample = acqui_optimizer(acqui_optimization, starting_point, Params::bayes_opt_bobase::bounded());
159  this->eval_and_add(sfun, new_sample);
160 
161  this->_update_stats(*this, afun);
162 
163  _model.add_sample(this->_samples.back(), this->_observations.back(), Params::bayes_opt_boptimizer::noise());
164 
165  if (Params::bayes_opt_boptimizer::hp_period() > 0
166  && (this->_current_iteration + 1) % Params::bayes_opt_boptimizer::hp_period() == 0)
167  _model.optimize_hyperparams();
168 
169  this->_current_iteration++;
170  this->_total_iterations++;
171  }
172  }
