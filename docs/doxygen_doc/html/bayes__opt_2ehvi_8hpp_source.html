102  template <typename EvalFunction>
103  void optimize(const EvalFunction& feval, bool reset = true)
104  {
105  this->_init(feval, FirstElem(), reset);
106 
107  acqui_optimizer_t inner_opt;
108 
109  while (this->_samples.size() == 0 || !this->_stop(*this, FirstElem())) {
110  this->template update_pareto_model<EvalFunction::dim_in>();
111  this->update_pareto_data();
112 
113  // copy in the ehvi structure to compute expected improvement
114  std::deque<individual*> pop;
115  for (auto x : this->pareto_data()) {
116  individual* ind = new individual;
117  ind->f[0] = std::get<1>(x)(0);
118  ind->f[1] = std::get<1>(x)(1);
119  ind->f[2] = 0;
120  pop.push_back(ind);
121  }
122 
123  auto acqui = acqui::Ehvi<Params, model_t>(
124  this->_models, pop,
125  Eigen::Vector3d(Params::bayes_opt_ehvi::x_ref(), Params::bayes_opt_ehvi::y_ref(), 0));
126 
127  // maximize with inner opt
128  using pair_t = std::pair<Eigen::VectorXd, double>;
129  pair_t init(Eigen::VectorXd::Zero(1), -std::numeric_limits<float>::max());
130 
131  auto body = [&](int i) -> pair_t {
132  auto x = this->pareto_data()[i];
133 
134  auto acqui_optimization =
135  [&](const Eigen::VectorXd& x, bool g) { return opt::no_grad(acqui(x)); };
136 
137  Eigen::VectorXd s = inner_opt(acqui_optimization, std::get<0>(x), true);
138  double hv = acqui(s);
139 
140  return std::make_pair(s, hv);
141  };
142 
143  auto comp = [](const pair_t& v1, const pair_t& v2) {
144  return v1.second > v2.second;
145  };
146 
147  auto m = tools::par::max(init, this->pareto_data().size(), body, comp);
148 
149  // take the best
150  Eigen::VectorXd new_sample = m.first;
151 
152  // delete pop
153  for (auto x : pop)
154  delete x;
155 
156  // add sample
157  this->add_new_sample(new_sample, feval(new_sample));
158  this->_update_stats(*this, FirstElem());
159  this->_current_iteration++;
160  this->_total_iterations++;
161  }
162  }
163  };
164  }
