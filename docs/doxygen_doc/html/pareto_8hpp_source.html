59  template <typename T>
60  static int dominate_flag(const T& i1, const T& i2)
61  {
62 
63  size_t nb_objs = i1.size();
64  assert(nb_objs);
65 
66  bool flag1 = false, flag2 = false;
67  for (unsigned i = 0; i < nb_objs; ++i) {
68  if (i1(i) > i2(i))
69  flag1 = true;
70  else if (i2(i) > i1(i))
71  flag2 = true;
72  }
73  if (flag1 && !flag2)
74  return 1;
75  else if (!flag1 && flag2)
76  return -1;
77  else
78  return 0;
79  }
