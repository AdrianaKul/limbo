414  void _optimize_hyperparams()
415  {
416  if (_optimize_init) {
417 
418  // Initialize parameter vector
419  _update_m();
420  _w_init = Eigen::VectorXd((_m + 1) * _dim_in + 2);
421 
422  // Initialize pseudo-inputs to a random subset of training inputs
423  Eigen::VectorXd positions = Eigen::VectorXd::LinSpaced(_samples.rows(), 0, _samples.rows() - 1);
424  std::random_shuffle(positions.data(), positions.data() + positions.size());
425  for (size_t i = 0; i < _m; ++i)
426  _w_init.segment(i * _dim_in, _dim_in) = _samples.row(positions[i]);
427 
428  // Initialize hyperparameters sensibly in log space
429  _w_init.segment(_m * _dim_in, _dim_in) = -2 * ((_samples.colwise().maxCoeff() - _samples.colwise().minCoeff()).array() / 2).log(); // -2*log((max(x)-min(x))'/2)
430  _w_init[(_m + 1) * _dim_in] = std::log(_observations_zm.array().pow(2).mean());
431  _w_init[(_m + 1) * _dim_in + 1] = std::log((_observations_zm.array().pow(2) / 4).mean());

458  opt::eval_t _likelihood_wp(Eigen::MatrixXd& xb, const Eigen::MatrixXd& b, const double& c, const double& sig,
459  const Eigen::MatrixXd& samples, const Eigen::MatrixXd& observations_zm, bool eval_grad, bool inverse) const
460  {
461  // Prepare xb and x
462  int n = samples.rows();
463  Eigen::MatrixXd idm = Eigen::MatrixXd::Identity(_m, _m);
464  Eigen::MatrixXd b_sqrt = b.array().sqrt();
465  xb = xb.cwiseProduct(b_sqrt.replicate(_m, 1));
466  Eigen::MatrixXd x = samples.cwiseProduct(b_sqrt.replicate(n, 1));
467 
468  // Construct Q
469  Eigen::MatrixXd Q = xb * xb.transpose();
470  Q = Eigen::MatrixXd(Q.diagonal().replicate(1, _m) + Q.diagonal().transpose().replicate(_m, 1) - 2 * Q);
471  Q = (Q.array() * -0.5).exp() * c;
472  Q += _del * idm;
473 
474  // NOTE: Need to be changed for the generic kernel construction
475  Eigen::MatrixXd k_m = x.cwiseProduct(x).rowwise().sum().transpose().replicate(_m, 1);
476  Eigen::MatrixXd k_n = xb.cwiseProduct(xb).rowwise().sum().replicate(1, n);
477  Eigen::MatrixXd K = -2 * xb * x.transpose() + k_m + k_n;
478  K = (K.array() * -0.5).exp() * c;
479 
480  // Resolve
481  Eigen::MatrixXd L = Q.llt().matrixL();
482  Eigen::TriangularView<Eigen::MatrixXd, Eigen::Lower> L_t = L.template triangularView<Eigen::Lower>();
483  Eigen::MatrixXd V = L_t.solve(K);
484 
485  Eigen::MatrixXd ep = 1 + (c - V.array().pow(2).colwise().sum().transpose()) / sig; // ep = 1 + (c-sum(V.^2)')/sig;
486  K = K.array() / ep.array().sqrt().transpose().replicate(_m, 1);
487  V = V.array() / ep.array().sqrt().transpose().replicate(_m, 1);
488  Eigen::MatrixXd y = observations_zm.array() / ep.array().sqrt();
489 
490  Eigen::MatrixXd Lm = (sig * idm + V * V.transpose()).llt().matrixL();
491  Eigen::TriangularView<Eigen::MatrixXd, Eigen::Lower> Lm_t = Lm.template triangularView<Eigen::Lower>();
492  Eigen::MatrixXd invLmV = Lm_t.solve(V);
493  Eigen::MatrixXd bet = invLmV * y;
494 
495  // Calculate likelihood
496  Eigen::MatrixXd fw = Lm.diagonal().array().log().colwise().sum() + (n - _m) / 2 * std::log(sig) + (y.transpose() * y - bet.transpose() * bet).array() / (2 * sig) + ep.array().log().sum() / 2 + 0.5 * n * std::log(2 * M_PI);
497 
498  if (!eval_grad) {
499  if (inverse)
500  return opt::no_grad(-fw(0));
501  else
502  return opt::no_grad(fw(0));
503  }
504 
505  // ** Derivates calculation - precomputations
506  Eigen::MatrixXd Lt = L * Lm;
507  Eigen::MatrixXd B1 = Lt.transpose().lu().solve(invLmV); // Lt is psd but Lt' it's not. We must use lu here. NOTE: Maybe it can be changed to solve the system with psd.
508  Eigen::MatrixXd b1 = Lt.transpose().lu().solve(bet);
509 
510  Eigen::MatrixXd invLV = L.transpose().lu().solve(V);
511  Eigen::MatrixXd invL = L.inverse();
512  Eigen::MatrixXd invQ = invL.transpose() * invL; // delete invL
513  Eigen::MatrixXd invLt = Lt.inverse();
514  Eigen::MatrixXd invA = invLt.transpose() * invLt; // delete invLt
515 
516  //NOTE: Throws exception after solving a transpose if multiplied directly with V, that's why its beign casted.
517  Eigen::MatrixXd mu = (((Eigen::MatrixXd)Lm.transpose().lu().solve(bet).transpose()) * V).transpose(); // NOTE: Check if the cast is correct here
518  Eigen::MatrixXd sumVsq = V.array().pow(2).colwise().sum().transpose();
519 
520  //bigsum = y.*(bet'*invLmV)'/sig - sum(invLmV.*invLmV)'/2 - (y.^2+mu.^2)/2/sig + 0.5
521  Eigen::MatrixXd bigsum = (((Eigen::MatrixXd)y.cwiseProduct((bet.transpose() * invLmV).transpose()) / sig) - ((Eigen::MatrixXd)invLmV.array().pow(2).colwise().sum().transpose() / 2) - ((Eigen::MatrixXd)(y.array().pow(2) + mu.array().pow(2)) / (2 * sig))).array() + 0.5;
522 
523  Eigen::MatrixXd TT = invLV * (invLV.transpose().cwiseProduct(bigsum.replicate(1, _m)));
524 
525  // Pseudo inputs and lengthscales
526  Eigen::MatrixXd dfxb(_m, _dim_in);
527  Eigen::MatrixXd dfb(_dim_in, 1);
528  for (size_t i = 0; i < _dim_in; i++) {
529  // dnnQ = dist(xb(:,i),xb(:,i)).*Q;
530  Eigen::MatrixXd r = xb.block(0, i, xb.rows(), 1);
531 
532  Eigen::MatrixXd dnnQ = _dist(xb.block(0, i, xb.rows(), 1), xb.block(0, i, xb.rows(), 1)).cwiseProduct(Q);
533  // dNnK = dist(-xb(:,i),-x(:,i)).*K;
534  Eigen::MatrixXd dNnK = _dist(xb.block(0, i, xb.rows(), 1) * -1, x.block(0, i, x.rows(), 1) * -1).cwiseProduct(K);
535 
536  // epdot = (dNnK.*invLV)*(-2/sig); epPmod = -sum(epdot)';
537  Eigen::MatrixXd epdot = dNnK.cwiseProduct(invLV) * (-2 / sig);
538  Eigen::MatrixXd epPmod = -1 * epdot.colwise().sum().transpose();
539 
540  // dfxb(:,i) = - b1.*(dNnK*(y-mu)/sig + dnnQ*b1) + sum((invQ - invA*sig).*dnnQ,2) + epdot*bigsum - 2/sig*sum(dnnQ.*TT,2);
541  dfxb.block(0, i, dfxb.rows(), 1) = b1.cwiseProduct((dNnK * (y - mu)) / sig + dnnQ * b1) * -1 + (invQ - invA * sig).cwiseProduct(dnnQ).rowwise().sum() + epdot * bigsum - (dnnQ.cwiseProduct(TT).rowwise().sum()) * (2 / sig);
542 
543  // dfb(i,1) = (((y-mu)'.*(b1'*dNnK))/sig + (epPmod.*bigsum)')*x(:,i);
544  dfb.block(i, 0, 1, 1) = ((y - mu).transpose().cwiseProduct(b1.transpose() * dNnK) / sig + epPmod.cwiseProduct(bigsum).transpose()) * x.block(0, i, x.rows(), 1);
545 
546  // Overwrite dNnK and write derivates
547  dNnK = Eigen::MatrixXd(dNnK.cwiseProduct(B1));
548  dfxb.block(0, i, dfxb.rows(), 1) += dNnK.rowwise().sum(); // dfxb(:,i) = dfxb(:,i) + sum(dNnK,2);
549  dfb.block(i, 0, 1, 1) -= dNnK.colwise().sum() * x.block(0, i, x.rows(), 1); // dfb(i,1) = dfb(i,1) - sum(dNnK,1)*x(:,i);
550 
551  dfxb.block(0, i, dfxb.rows(), 1) *= std::sqrt(b(i)); // dfxb(:,i) = dfxb(:,i)*sqrt(b(i));
552 
553  dfb.block(i, 0, 1, 1) /= std::sqrt(b(i)); // dfb(i,1) = dfb(i,1)/sqrt(b(i));
554  dfb.block(i, 0, 1, 1) += (dfxb.block(0, i, dfxb.rows(), 1).transpose() * // dfb(i,1) = dfb(i,1) + (dfxb(:,i)'*xb(:,i))/b(i);
555  xb.block(0, i, xb.rows(), 1))
556  / b(i);
557  dfb.block(i, 0, 1, 1) *= std::sqrt(b(i)) / 2; // dfb(i,1) = dfb(i,1)*sqrt(b(i))/2;
558  }
559 
560  // Size
561  // epc = (c./ep - sumVsq - del*sum((invLV).^2)')/sig
562  Eigen::MatrixXd epc = (ep.cwiseInverse() * c - sumVsq - _del * (((Eigen::MatrixXd)invLV.array().pow(2)).colwise().sum().transpose())) / sig;
563 
564  // dfc = (m + del*trace(invQ-sig*invA) - sig*sum(sum(invA.*Q')))/2 -
565  // mu'*(y-mu)/sig + (b1'*(Q-del*eye(m))*b1)/2 + epc'*bigsum;
566  Eigen::MatrixXd dfc = (_m + _del * (invQ - sig * invA).trace() - sig * (invA.cwiseProduct(Q.transpose()).sum())) / 2 - ((mu.transpose() * (y - mu)) / sig).array() + ((b1.transpose() * (Q - _del * idm) * b1) / 2).array() + (epc.transpose() * bigsum).array();
567 
568  // Noise - dfsig = sum(bigsum./ep);
569  Eigen::MatrixXd dfsig = bigsum.cwiseQuotient(ep).colwise().sum();
570 
571  // Assemble the result
572  Eigen::VectorXd dfw(dfxb.size() + dfb.size() + dfc.size() + dfsig.size());
573  dfw.segment(0, dfxb.size()) = Eigen::Map<Eigen::MatrixXd>(dfxb.data(), 1, dfxb.size()).transpose();
574  dfw.segment(dfxb.size(), dfb.size()) = dfb;
575  dfw.segment(dfxb.size() + dfb.size(), dfc.size()) = dfc;
576  dfw.segment(dfxb.size() + dfb.size() + dfc.size(), dfsig.size()) = dfsig;
577 
578  if (inverse) {
579  Eigen::VectorXd rdfw = dfw.array() * -1;
580  return {-fw(0), rdfw}; // limbo maximizes instead of minimizing
581  }
582  else {
583  return {fw(0), dfw};
584  }
585  }

617  Eigen::MatrixXd _compute_kernel_matrix(const Eigen::MatrixXd& p_1, const Eigen::MatrixXd& p_2) const
618  {
619  Eigen::MatrixXd K;
620  int n_1 = p_1.rows();
621  int n_2 = p_2.rows();
622 
623  Eigen::MatrixXd bx_1 = _b.transpose().array().sqrt().replicate(n_1, 1);
624  Eigen::MatrixXd bx_2 = _b.transpose().array().sqrt().replicate(n_2, 1);
625 
626  Eigen::MatrixXd x_1 = p_1.cwiseProduct(bx_1);
627  Eigen::MatrixXd x_2 = p_2.cwiseProduct(bx_2);
628 
629  K = (-2 * x_1 * x_2.transpose()).array() + x_2.array().pow(2).rowwise().sum().transpose().replicate(n_1, 1) + x_1.array().pow(2).rowwise().sum().replicate(1, n_2);
630  K = _c * (K * -0.5).array().exp();
631 
632  return K;
633  }
