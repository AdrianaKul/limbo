116  std::vector<Eigen::VectorXd> _scalarize_obs(const std::vector<Eigen::VectorXd>& observations)
117  {
118  Eigen::VectorXd lambda = tools::random_vector(_nb_objs);
119  double sum = lambda.sum();
120  lambda = lambda / sum;
121  // scalarize (Tchebycheff)
122  std::vector<Eigen::VectorXd> scalarized;
123  for (auto x : observations) {
124  double y = (lambda.array() * x.array()).maxCoeff();
125  double s = (lambda.array() * x.array()).sum();
126  auto v = tools::make_vector(y + Params::model_gp_parego::rho() * s);
127  scalarized.push_back(v);
128  }
129  return scalarized;
130  }
