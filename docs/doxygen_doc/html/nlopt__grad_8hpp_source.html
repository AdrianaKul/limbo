107 
108  opt.set_max_objective(nlopt_func<F>, (void*)&f);
109 
110  std::vector<double> x(dim);

112 
113  opt.set_maxeval(Params::opt_nloptgrad::iterations());
114 
115  if (bounded) {
116  opt.set_lower_bounds(std::vector<double>(dim, 0));
117  opt.set_upper_bounds(std::vector<double>(dim, 1));
118  }
119 
120  double max;
121 
122  try {
123  opt.optimize(x, max);
124  }

134  // In theory it's ok to ignore this error

136  }
137

144  {

146  Eigen::VectorXd params = Eigen::VectorXd::Map(x.data(), x.size());
147  double v;
