95  // We expect the input parameters to be in log-space
96  void set_h_params(const Eigen::VectorXd& p)
97  {
98  _h_params = p;
99  for (size_t i = 0; i < _input_dim; ++i)
100  _ell(i) = std::exp(p(i));
101  for (size_t j = 0; j < (unsigned int)Params::kernel_squared_exp_ard::k(); ++j)
102  for (size_t i = 0; i < _input_dim; ++i)
103  _A(i, j) = std::exp(p((j + 1) * _input_dim + i));
104  }

106  Eigen::VectorXd grad(const Eigen::VectorXd& x1, const Eigen::VectorXd& x2) const
107  {
108  if (Params::kernel_squared_exp_ard::k() > 0) {
109  Eigen::VectorXd grad = Eigen::VectorXd::Zero(this->h_params_size());
110  Eigen::MatrixXd K = (_A * _A.transpose());
111  K.diagonal() += (Eigen::MatrixXd)(_ell.array().inverse().square());
112  double z = ((x1 - x2).transpose() * K * (x1 - x2)).norm();
113  double k = _sf2 * std::exp(-0.5 * z);
114 
115  grad.head(_input_dim) = (x1 - x2).cwiseQuotient(_ell).array().square() * k;
116  Eigen::MatrixXd G = -k * (x1 - x2) * (x1 - x2).transpose() * _A;
117  for (size_t j = 0; j < Params::kernel_squared_exp_ard::k(); ++j)
118  grad.segment((1 + j) * _input_dim, _input_dim) = G.col(j);
119 
120  return grad;
121  }
122  else {
123  Eigen::VectorXd grad(_input_dim);
124  Eigen::VectorXd z = (x1 - x2).cwiseQuotient(_ell).array().square();
125  double k = _sf2 * std::exp(-0.5 * z.sum());
126  grad.head(_input_dim) = z * k;
127  return grad;
128  }
129  }

131  double operator()(const Eigen::VectorXd& x1, const Eigen::VectorXd& x2) const
132  {
133  assert(x1.size() == _ell.size());
134  double z;
135  if (Params::kernel_squared_exp_ard::k() > 0) {
136  Eigen::MatrixXd K = (_A * _A.transpose());
137  K.diagonal() += (Eigen::MatrixXd)(_ell.array().inverse().square());
138  z = ((x1 - x2).transpose() * K * (x1 - x2)).norm();
139  }
140  else {

142  }
143  return _sf2 * std::exp(-0.5 * z);
144  }
