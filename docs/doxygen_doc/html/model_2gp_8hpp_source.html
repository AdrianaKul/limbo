320  void _compute_incremental_kernel()
321  {
322  // Incremental LLT
323  // This part of the code is inpired from the Bayesopt Library (cholesky_add_row function).
324  // However, the mathematical fundations can be easily retrieved by detailling the equations of the
325  // extended L matrix that produces the desired kernel.
326 
327  size_t n = _samples.size();
328  _kernel.conservativeResize(n, n);
329 
330  for (size_t i = 0; i < n; ++i) {
331  _kernel(i, n - 1) = _kernel_function(_samples[i], _samples[n - 1]) + ((i == n - 1) ? _noises[i] : 0); // noise only on the diagonal
332  _kernel(n - 1, i) = _kernel(i, n - 1);
333  }
334 
335  _matrixL.conservativeResizeLike(Eigen::MatrixXd::Zero(n, n));
336 
337  double L_j;
338  for (size_t j = 0; j < n - 1; ++j) {
339  L_j = _kernel(n - 1, j) - (_matrixL.block(j, 0, 1, j) * _matrixL.block(n - 1, 0, 1, j).transpose())(0, 0);
340  _matrixL(n - 1, j) = (L_j) / _matrixL(j, j);
341  }
342 
343  L_j = _kernel(n - 1, n - 1) - (_matrixL.block(n - 1, 0, 1, n - 1) * _matrixL.block(n - 1, 0, 1, n - 1).transpose())(0, 0);
344  _matrixL(n - 1, n - 1) = sqrt(L_j);
345 
346  this->_compute_alpha();
347  }

356  void _compute_bl_kernel()
357  {
358  Eigen::MatrixXd A1 = Eigen::MatrixXd::Identity(this->_samples.size(), this->_samples.size());
359  _matrixL.template triangularView<Eigen::Lower>().solveInPlace(A1);
360  _matrixL.template triangularView<Eigen::Lower>().transpose().solveInPlace(A1);
361 
362  _inv_bl_kernel.resize(_samples.size() + _bl_samples.size(),
363  _samples.size() + _bl_samples.size());
364 
365  Eigen::MatrixXd B(_samples.size(), _bl_samples.size());
366  for (size_t i = 0; i < _samples.size(); i++)
367  for (size_t j = 0; j < _bl_samples.size(); ++j)
368  B(i, j) = _kernel_function(_samples[i], _bl_samples[j]);
369 
370  Eigen::MatrixXd D(_bl_samples.size(), _bl_samples.size());
371  for (size_t i = 0; i < _bl_samples.size(); i++)
372  for (size_t j = 0; j < _bl_samples.size(); ++j)
373  D(i, j) = _kernel_function(_bl_samples[i], _bl_samples[j]) + ((i == j) ? _noises_bl[i] : 0);
374 
375  Eigen::MatrixXd comA = (D - B.transpose() * A1 * B);
376  Eigen::LLT<Eigen::MatrixXd> llt_bl(comA);
377  Eigen::MatrixXd comA1 = Eigen::MatrixXd::Identity(_bl_samples.size(), _bl_samples.size());
378  llt_bl.matrixL().solveInPlace(comA1);
379  llt_bl.matrixL().transpose().solveInPlace(comA1);
380 
381  // fill the matrix block wise
382  _inv_bl_kernel.block(0, 0, _samples.size(), _samples.size()) = A1 + A1 * B * comA1 * B.transpose() * A1;
383  _inv_bl_kernel.block(0, _samples.size(), _samples.size(),
384  _bl_samples.size())
385  = -A1 * B * comA1;
386  _inv_bl_kernel.block(_samples.size(), 0, _bl_samples.size(),
387  _samples.size())
388  = _inv_bl_kernel.block(0, _samples.size(), _samples.size(),
389  _bl_samples.size())
390  .transpose();
391  _inv_bl_kernel.block(_samples.size(), _samples.size(), _bl_samples.size(),
392  _bl_samples.size())
393  = comA1;
394  }
