79  void compute(const std::vector<Eigen::VectorXd>& samples,
80  const std::vector<Eigen::VectorXd>& observations,
81  const Eigen::VectorXd& noises, bool compute_kernel = true)
82  {
83  assert(samples.size() != 0);
84  assert(observations.size() != 0);
85  assert(samples.size() == observations.size());
86 
87  if (_dim_in != samples[0].size()) {
88  _dim_in = samples[0].size();
89  _kernel_function = KernelFunction(_dim_in); // the cost of building a functor should be relatively low
90  }
91 
92  if (_dim_out != observations[0].size()) {
93  _dim_out = observations[0].size();
94  _mean_function = MeanFunction(_dim_out); // the cost of building a functor should be relatively low
95  }
96 
97  _samples = samples;
98 
99  _observations.resize(observations.size(), _dim_out);
100  for (int i = 0; i < _observations.rows(); ++i)
101  _observations.row(i) = observations[i];
102 
103  _mean_observation = _observations.colwise().mean();
104 
105  _noises = noises;
106 
107  this->_compute_obs_mean();
108  if (compute_kernel)
109  this->_compute_full_kernel();
110  }

293  void _compute_obs_mean()
294  {
295  _mean_vector.resize(_samples.size(), _dim_out);
296  for (int i = 0; i < _mean_vector.rows(); i++)
297  _mean_vector.row(i) = _mean_function(_samples[i], *this);

301  void _compute_full_kernel()
302  {
303  size_t n = _samples.size();
304  _kernel.resize(n, n);
305 
306  // O(n^2) [should be negligible]
307  for (size_t i = 0; i < n; i++)
308  for (size_t j = 0; j <= i; ++j)
309  _kernel(i, j) = _kernel_function(_samples[i], _samples[j]) + ((i == j) ? _noises[i] : 0); // noise only on the diagonal
310 
311  for (size_t i = 0; i < n; i++)
312  for (size_t j = 0; j < i; ++j)
313  _kernel(j, i) = _kernel(i, j);
314 
315  // O(n^3)
316  _matrixL = Eigen::LLT<Eigen::MatrixXd>(_kernel).matrixL();
317 
318  this->_compute_alpha();
319  }

321  void _compute_incremental_kernel()
322  {
323  // Incremental LLT
324  // This part of the code is inpired from the Bayesopt Library (cholesky_add_row function).
325  // However, the mathematical fundations can be easily retrieved by detailling the equations of the
326  // extended L matrix that produces the desired kernel.
327 
328  size_t n = _samples.size();
329  _kernel.conservativeResize(n, n);
330 
331  for (size_t i = 0; i < n; ++i) {
332  _kernel(i, n - 1) = _kernel_function(_samples[i], _samples[n - 1]) + ((i == n - 1) ? _noises[i] : 0); // noise only on the diagonal
333  _kernel(n - 1, i) = _kernel(i, n - 1);
334  }
335 
336  _matrixL.conservativeResizeLike(Eigen::MatrixXd::Zero(n, n));
337 
338  double L_j;
339  for (size_t j = 0; j < n - 1; ++j) {
340  L_j = _kernel(n - 1, j) - (_matrixL.block(j, 0, 1, j) * _matrixL.block(n - 1, 0, 1, j).transpose())(0, 0);
341  _matrixL(n - 1, j) = (L_j) / _matrixL(j, j);
342  }
343 
344  L_j = _kernel(n - 1, n - 1) - (_matrixL.block(n - 1, 0, 1, n - 1) * _matrixL.block(n - 1, 0, 1, n - 1).transpose())(0, 0);
345  _matrixL(n - 1, n - 1) = sqrt(L_j);
346 
347  this->_compute_alpha();
348  }

350  void _compute_alpha()
351  {
352  // alpha = K^{-1} * this->_obs_mean;
353  Eigen::TriangularView<Eigen::MatrixXd, Eigen::Lower> triang = _matrixL.template triangularView<Eigen::Lower>();
354  _alpha = triang.solve(_obs_mean);
355  triang.adjoint().solveInPlace(_alpha);
356  }

363  double _sigma(const Eigen::VectorXd& v, const Eigen::VectorXd& k) const
364  {
365  Eigen::VectorXd z = _matrixL.triangularView<Eigen::Lower>().solve(k);
366  double res = _kernel_function(v, v) - z.dot(z);
367 
368  return (res <= std::numeric_limits<double>::epsilon()) ? 0 : res;
369  }
