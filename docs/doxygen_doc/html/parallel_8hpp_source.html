152  template <typename T, typename F, typename C>
153  T max(const T& init, int num_steps, const F& f, const C& comp)
154  {
155 #ifdef USE_TBB
156  auto body = [&](const tbb::blocked_range<size_t>& r, T current_max) -> T {
157  // clang-format off
158  for (size_t i = r.begin(); i != r.end(); ++i)
159  {
160  T v = f(i);
161  if (comp(v, current_max))
162  current_max = v;
163  }
164  return current_max;
165  // clang-format on
166  };
167  auto joint = [&](const T& p1, const T& p2) -> T {
168  // clang-format off
169  if (comp(p1, p2))
170  return p1;
171  return p2;
172  // clang-format on
173  };
174  return tbb::parallel_reduce(tbb::blocked_range<size_t>(0, num_steps), init,
175  body, joint);
176 #else
177  T current_max = init;
178  for (int i = 0; i < num_steps; ++i) {
179  T v = f(i);
180  if (comp(v, current_max))
181  current_max = v;
182  }
183  return current_max;
184 #endif
185  }
