86  struct Cmaes {
87  public:
88  template <typename F>
89  Eigen::VectorXd operator()(const F& f, const Eigen::VectorXd& init, double bounded) const
90  {
91  size_t dim = init.size();
92 
93  // wrap the function
94  libcmaes::FitFunc f_cmaes = [&](const double* x, const int n) {
95  Eigen::Map<const Eigen::VectorXd> m(x, n);
96  // remember that our optimizers maximize
97  return -eval(f, m);
98  };
99 
100  if (bounded)
101  return _opt_bounded(f_cmaes, dim, init);
102  else
103  return _opt_unbounded(f_cmaes, dim, init);
104  }
105 
106  private:
107  // F is a CMA-ES style function, not our function
108  template <typename F>
109  Eigen::VectorXd _opt_unbounded(F& f_cmaes, int dim, const Eigen::VectorXd& init) const
110  {
111  using namespace libcmaes;
112  // initial step-size, i.e. estimated initial parameter error.
113  double sigma = 0.5;
114  std::vector<double> x0(init.data(), init.data() + init.size());
115 
116  CMAParameters<> cmaparams(x0, sigma);
117  _set_common_params(cmaparams, dim);
118  // used by restart I think
119  cmaparams.set_x0(-1.0, 1.0);
120 
121  // the optimization itself
122  CMASolutions cmasols = cmaes<>(f_cmaes, cmaparams);
123  return cmasols.get_best_seen_candidate().get_x_dvec();
124  }

126  // F is a CMA-ES style function, not our function
127  template <typename F>
128  Eigen::VectorXd _opt_bounded(F& f_cmaes, int dim, const Eigen::VectorXd& init) const
129  {
130  using namespace libcmaes;
131  // create the parameter object
132  // boundary_transformation
133  double lbounds[dim], ubounds[dim]; // arrays for lower and upper parameter bounds, respectively
134  for (int i = 0; i < dim; i++) {
135  lbounds[i] = 0.0;
136  ubounds[i] = 1.0;
137  }
138  GenoPheno<pwqBoundStrategy> gp(lbounds, ubounds, dim);
139  // initial step-size, i.e. estimated initial parameter error.
140  // we suppose we are optimizing on [0, 1], but we have no idea where to start
141  double sigma = 0.5;
142  std::vector<double> x0(init.data(), init.data() + init.size());
143  // -1 for automatically decided lambda, 0 is for random seeding of the internal generator.
144  CMAParameters<GenoPheno<pwqBoundStrategy>> cmaparams(dim, &x0.front(), sigma, -1, 0, gp);
145  _set_common_params(cmaparams, dim);
146  // used by restart I think
147  cmaparams.set_x0(0, 1.0);
148 
149  // the optimization itself
150  CMASolutions cmasols = cmaes<GenoPheno<pwqBoundStrategy>>(f_cmaes, cmaparams);
151  //cmasols.print(std::cout, 1, gp);
152  //to_f_representation
153  return gp.pheno(cmasols.get_best_seen_candidate().get_x_dvec());
154  }

156  template <typename P>
157  void _set_common_params(P& cmaparams, int dim) const
158  {
159  using namespace libcmaes;
160 
161  // set multi-threading to true
162  cmaparams.set_mt_feval(true);
163  // aCMAES should be the best choice
164  // [see: https://github.com/beniz/libcmaes/wiki/Practical-hints ]
165  // but we want the restart -> aIPOP_CMAES
166  cmaparams.set_algo(aIPOP_CMAES);
167  cmaparams.set_restarts(Params::opt_cmaes::restarts());
168  // if no max fun evals provided, we compute a recommended value
169  size_t max_evals = Params::opt_cmaes::max_fun_evals() < 0
170  ? (900.0 * (dim + 3.0) * (dim + 3.0))
171  : Params::opt_cmaes::max_fun_evals();
172  cmaparams.set_max_fevals(max_evals);
173  // max iteration is here only for security
174  cmaparams.set_max_iter(100000);
175  // we do not know if what is the actual maximum / minimum of the function
176  // therefore we deactivate this stopping criterion
177  cmaparams.set_stopping_criteria(FTARGET, false);
178  }
179  };
180  }
