149  template <typename StateFunction, typename AggregatorFunction = FirstElem>
150  void optimize(const StateFunction& sfun, const AggregatorFunction& afun = AggregatorFunction(), bool reset = true)
151  {
152  _nb_constraints = StateFunction::nb_constraints;
153  _dim_out = StateFunction::dim_out;
154 
155  this->_init(sfun, afun, reset);
156 
157  if (!this->_observations.empty()) {
158  _split_observations();
159  _model.compute(this->_samples, _obs[0], Eigen::VectorXd::Constant(_obs[0].size(), Params::bayes_opt_cboptimizer::noise()));
160  if (_nb_constraints > 0)
161  _constraint_model.compute(this->_samples, _obs[1], Eigen::VectorXd::Constant(_obs[1].size(), Params::bayes_opt_cboptimizer::noise()));
162  }
163  else {
164  _model = model_t(StateFunction::dim_in, StateFunction::dim_out);
165  if (_nb_constraints > 0)
166  _constraint_model = constraint_model_t(StateFunction::dim_in, _nb_constraints);
167  }
168 
169  acqui_optimizer_t acqui_optimizer;
170 
171  while (!this->_stop(*this, afun)) {
172  acquisition_function_t acqui(_model, _constraint_model, this->_current_iteration);
173 
174  auto acqui_optimization =
175  [&](const Eigen::VectorXd& x, bool g) { return acqui(x,afun,g); };
176  Eigen::VectorXd starting_point = tools::random_vector(StateFunction::dim_in, Params::bayes_opt_cboptimizer::bounded());
177  Eigen::VectorXd new_sample = acqui_optimizer(acqui_optimization, starting_point, Params::bayes_opt_cboptimizer::bounded());
178  this->eval_and_add(sfun, new_sample);
179 
180  this->_update_stats(*this, afun);
181 
182  _model.add_sample(this->_samples.back(), _obs[0].back(), Params::bayes_opt_cboptimizer::noise());
183  if (_nb_constraints > 0)
184  _constraint_model.add_sample(this->_samples.back(), _obs[1].back(), Params::bayes_opt_cboptimizer::noise());
185 
186  if (Params::bayes_opt_cboptimizer::hp_period() > 0
187  && (this->_current_iteration + 1) % Params::bayes_opt_cboptimizer::hp_period() == 0) {
188  _model.optimize_hyperparams();
189  if (_nb_constraints > 0)
190  _constraint_model.optimize_hyperparams();
191  }
192 
193  this->_current_iteration++;
194  this->_total_iterations++;
195  }
196  }

200  const Eigen::VectorXd& best_observation(const AggregatorFunction& afun = AggregatorFunction()) const
201  {
202  _dim_out = _model.dim_out();
203  _split_observations();
204 
205  std::vector<Eigen::VectorXd> obs = _feasible_observations();
206  if (obs.size() > 0)

252  void _split_observations() const
253  {
254  _obs.clear();
255  _obs.resize(2);
256 
257  for (size_t i = 0; i < this->_observations.size(); ++i) {
258  assert(size_t(this->_observations[i].size()) == _dim_out + _nb_constraints);
259 
260  Eigen::VectorXd vec_obj(_dim_out);
261  for (size_t j = 0; j < _dim_out; ++j)
262  vec_obj(j) = this->_observations[i](j);
263  _obs[0].push_back(vec_obj);
264 
265  if (_nb_constraints > 0) {
266  Eigen::VectorXd vec_con(_nb_constraints);
267  for (int j = _dim_out, ind = 0; j < this->_observations[i].size(); ++j, ++ind)
268  vec_con(ind) = this->_observations[i](j);
269  _obs[1].push_back(vec_con);
270  }
271  }
272  }
