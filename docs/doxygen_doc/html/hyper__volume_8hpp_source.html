64  template <typename BO, typename AggregatorFunction>
65  void operator()(const BO& bo, const AggregatorFunction&)
66  {
67  if (bo.observations().empty())
68  return;
69  if (!bo.stats_enabled())
70  return;
71  // convert the data to C arrays
72  double** data = new double* [bo.observations().size()];
73  for (size_t i = 0; i < bo.observations().size(); ++i) {
74  size_t dim = bo.observations()[i].size();
75  data[i] = new double[dim];
76  for (size_t k = 0; k < dim; ++k)
77  data[i][k] = bo.observations()[i](k) + Params::stat_hyper_volume::ref(k);
78  }
79  // call the hypervolume by Zitzler
80  int noObjectives = bo.observations()[0].size();
81  int redSizeFront1 = FilterNondominatedSet(data, bo.observations().size(), noObjectives);
82  double hv = CalculateHypervolume(data, redSizeFront1, noObjectives);
83 
84  // write
85  this->_create_log_file(bo, "hypervolume.dat");
86  (*this->_log_file) << bo.current_iteration() << "\t" << hv << std::endl;
87 
88  // free data
89  for (size_t i = 0; i < bo.observations().size(); ++i)
90  delete[] data[i];
91  delete[] data;
92  }
93  };
94  }
95  }
