78  template <typename F>
79  Eigen::VectorXd operator()(const F& f, const Eigen::VectorXd& init, bool bounded) const
80  {
81  // params
82  size_t param_dim = init.size();
83  double delta0 = 0.1;
84  double deltamin = 1e-6;
85  double deltamax = 50;
86  double etaminus = 0.5;
87  double etaplus = 1.2;
88  double eps_stop = 0.0;
89 
90  Eigen::VectorXd delta = Eigen::VectorXd::Ones(param_dim) * delta0;
91  Eigen::VectorXd grad_old = Eigen::VectorXd::Zero(param_dim);
92  Eigen::VectorXd params = init;
93 
94  if (bounded) {
95  for (int j = 0; j < params.size(); j++) {
96  if (params(j) < 0)
97  params(j) = 0;
98  if (params(j) > 1)
99  params(j) = 1;
100  }
101  }
102 
103  Eigen::VectorXd best_params = params;
104  double best = log(0);
105 
106  for (int i = 0; i < Params::opt_rprop::iterations(); ++i) {
107  auto perf = opt::eval_grad(f, params);
108  double lik = opt::fun(perf);
109  if (lik > best) {
110  best = lik;
111  best_params = params;
112  }
113  Eigen::VectorXd grad = -opt::grad(perf);
114  grad_old = grad_old.cwiseProduct(grad);
115 
116  for (int j = 0; j < grad_old.size(); ++j) {
117  if (grad_old(j) > 0) {
118  delta(j) = std::min(delta(j) * etaplus, deltamax);
119  }
120  else if (grad_old(j) < 0) {
121  delta(j) = std::max(delta(j) * etaminus, deltamin);
122  grad(j) = 0;
123  }
124  params(j) += -tools::signum(grad(j)) * delta(j);
125 
126  if (bounded && params(j) < 0)
127  params(j) = 0;
128  if (bounded && params(j) > 1)
129  params(j) = 1;
130  }
131 
132  grad_old = grad;
133  if (grad_old.norm() < eps_stop)
134  break;
135  }
136 
137  return best_params;
138  }
139  };
140  }
