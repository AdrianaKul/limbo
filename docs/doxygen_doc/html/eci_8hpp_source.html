75  template <typename AggregatorFunction>
76  opt::eval_t operator()(const Eigen::VectorXd& v, const AggregatorFunction& afun, bool gradient)
77  {
78  assert(!gradient);
79 
80  Eigen::VectorXd mu;
81  double sigma_sq;
82  std::tie(mu, sigma_sq) = _model.query(v);
83  double sigma = std::sqrt(sigma_sq);
84 
85  // If \sigma(x) = 0 or we do not have any observation yet we return 0
86  if (sigma < 1e-10 || _model.samples().size() < 1)
87  return opt::no_grad(0.0);
88 
89  // Compute expected constrained improvement
90  // First find the best (predicted) observation so far -- if needed
91  if (_nb_samples != _model.nb_samples()) {
92  std::vector<double> rewards;
93  for (auto s : _model.samples()) {
94  rewards.push_back(afun(_model.mu(s)));
95  }
96 
97  _nb_samples = _model.nb_samples();
98  _f_max = *std::max_element(rewards.begin(), rewards.end());
99  }
100  // Calculate Z and \Phi(Z) and \phi(Z)
101  double X = afun(mu) - _f_max - Params::acqui_eci::jitter();
102  double Z = X / sigma;
103  double phi = std::exp(-0.5 * std::pow(Z, 2.0)) / std::sqrt(2.0 * M_PI);
104  double Phi = 0.5 * std::erfc(-Z / std::sqrt(2));
105 
106  return opt::no_grad(_pf(v, afun) * (X * Phi + sigma * phi));
107  }
108 
109  protected:
110  const Model& _model;
111  const ConstraintModel& _constraint_model;
112  int _nb_samples;
113  double _f_max;
114 
115  template <typename AggregatorFunction>
116  double _pf(const Eigen::VectorXd& v, const AggregatorFunction& afun) const
117  {
118  Eigen::VectorXd mu;
119  double sigma_sq;
120  std::tie(mu, sigma_sq) = _constraint_model.query(v);
121  double sigma = std::sqrt(sigma_sq);
122 
123  if (sigma < 1e-10 || _constraint_model.samples().size() < 1)
124  return 1.0;
125 
126  double Z = (afun(mu) - 1.0) / sigma;
127  double Phi = 0.5 * std::erfc(-Z / std::sqrt(2));
128 
129  return Phi;
130  }
131  };
